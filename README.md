	  user:
	    name: user
	    password: password
	
	eureka: 
	  client:
	    registerWithEureka: false
	    fetchRegistry: false
	  server:
	    wait-time-in-ms-when-sync-empty: 0

	@EnableConfigServer

	  client:
	    registry-fetch-interval-seconds: 5
	    serviceUrl:
	       defaultZone: http://user:password@${domain.name:localhost}:8761/eureka/
	  
	spring:  
	  cloud:
